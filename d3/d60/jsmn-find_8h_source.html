146static int
147_jsmnf_get_pairs(const char js[], jsmntok_t *tok, size_t ntoks, jsmnf *head)
148{
149 int offset = 0;
150
151 if (!ntoks) return 0;
152
153 switch (tok->type) {
154 case JSMN_OBJECT: {
155 jsmnf *curr;
156 int ret;
157 int i;
158
159 for (i = 0; i < tok->size; ++i) {
160 curr = _jsmnf_init();
161 curr->idx = i;
162 curr->key = tok + 1 + offset;
163
164 ret = _jsmnf_get_pairs(js, curr->key, ntoks - offset, curr);
165 if (ret < 0) return ret;
166
167 offset += ret;
168
169 if (curr->key->size > 0) {
170 curr->val = tok + 1 + offset;
171
172 ret = _jsmnf_get_pairs(js, curr->val, ntoks - offset, curr);
173 if (ret < 0) return ret;
174
175 offset += ret;
176 }
177
178 HASH_ADD_KEYPTR(hh, head->child, js + curr->key->start,
179 curr->key->end - curr->key->start, curr);
180 }
181 } break;
182 case JSMN_ARRAY: {
183 jsmnf *curr;
184 int ret;
185 int i;
186
187 for (i = 0; i < tok->size; ++i) {
188 curr = _jsmnf_init();
189 curr->idx = i;
190 curr->val = tok + 1 + offset;
191
192 ret = _jsmnf_get_pairs(js, curr->val, ntoks - offset, curr);
193 if (ret < 0) return ret;
194
195 offset += ret;
196
197 HASH_ADD_INT(head->child, idx, curr);
198 }
199 } break;
200 case JSMN_STRING:
201 case JSMN_PRIMITIVE:
202 break;
203 case JSMN_UNDEFINED:
204 default:
205 fprintf(stderr, "Unexpected key: %.*s\n", tok->end - tok->start,
206 js + tok->start);
207 return -1;
208 }
209 return offset + 1;
210}

212int
213jsmnf_start(jsmnf *root, const char js[], size_t size)
214{
215 struct _jsmnroot *r = (struct _jsmnroot *)root;
216 jsmn_parser parser;
217 int ret;
218
219 /* Prepare parser */
220 jsmn_init(&parser);
221 while (1) {
222 ret = jsmn_parse(&parser, js, size, root->val, r->real_ntoks);
223
224 if (ret >= 0) {
225 r->ntoks = parser.toknext;
226 ret = _jsmnf_get_pairs(js, root->val, r->ntoks, root);
227 break;
228 }
229 else {
230 if (ret != JSMN_ERROR_NOMEM) {
231 break;
232 }
233 else {
234 size_t new_ntoks = r->real_ntoks * 2;
235 void *tmp;
236
237 tmp = realloc(root->val, new_ntoks * sizeof *root->val);
238 if (!tmp) return JSMN_ERROR_NOMEM;
239
240 r->real_ntoks = new_ntoks;
241 root->val = tmp;
242 }
243 }
244 }
245 return ret;
246}

248jsmnf *
249jsmnf_find(jsmnf *head, const char key[], size_t size)
250{
251 jsmnf *found = NULL;
252
253 if (!key || !head) return NULL;
254
255 if (JSMN_OBJECT == head->val->type) {
256 HASH_FIND(hh, head->child, key, size, found);
257 }
258 else if (JSMN_ARRAY == head->val->type) {
259 char *endptr;
260 int idx = (int)strtol(key, &endptr, 10);
261
262 if (endptr == key) return NULL;
263
264 HASH_FIND_INT(head->child, &idx, found);
265 }
266 return found;
267}

284static int
285read_4_digits(char **str_p, const char *src_end, unsigned *x)
286{
287 char *str = *str_p;
288 char buf[5] = { 0 };
289 unsigned v;
290 int i;
291
292 if (src_end - str < 4) return 0;
293
294 for (i = 0; i < 4; i++) {
295 char c = str[i];
296
297 buf[i] = c;
298 if (isxdigit(c)) continue;
299
300 return 0;
301 }
302
303 sscanf(buf, "%x", &v);
304
305 *x = v;
306 *str_p = str + 4;
307
308 return 1;
309}

311#define UTF16_IS_FIRST_SURROGATE(c) \
312 (0xD800 <= (unsigned)c && (unsigned)c <= 0xDBFF)
313#define UTF16_IS_SECOND_SURROGATE(c) \
314 (0xDC00 <= (unsigned)c && (unsigned)c <= 0xDFFF)
315#define UTF16_JOIN_SURROGATE(c1, c2) \
316 (((((unsigned long)c1 & 0x3FF) << 10) | ((unsigned)c2 & 0x3FF)) + 0x10000)
317#define UTF8_IS_VALID(c) \
318 (((unsigned long)c <= 0x10FFFF) \
319 && ((unsigned long)c < 0xD800 || (unsigned long)c > 0xDFFF))
320#define UTF8_IS_TRAIL(c) (((unsigned char)c & 0xC0) == 0x80)
321#define UTF_ILLEGAL 0xFFFFFFFFu

325{
326 if (c < 128) return 0;

347/* See RFC 3629
348 Based on: http://www.w3.org/International/questions/qa-forms-utf-8 */
349static unsigned long
350utf8_next(char **p, char *e, int html)
351{
352 unsigned char lead, tmp;
353 int trail_size;
354 unsigned long c;
355
356 if (*p == e) return UTF_ILLEGAL;
357
358 lead = **p;
359 (*p)++;
360
361 /* First byte is fully validated here */
362 trail_size = utf8_trail_length(lead);
363
364 if (trail_size < 0) return UTF_ILLEGAL;
365
366 /* Ok as only ASCII may be of size = 0 also optimize for ASCII text */
367 if (trail_size == 0) {
368 if (!html || (lead >= 0x20 && lead != 0x7F) || lead == 0x9
369 || lead == 0x0A || lead == 0x0D)
370 return lead;
371 return UTF_ILLEGAL;
372 }
373
374 c = lead & ((1 << (6 - trail_size)) - 1);
375
376 /* Read the rest */
377 switch (trail_size) {
378 case 3:
379 if (*p == e) return UTF_ILLEGAL;
380 tmp = **p;
381 (*p)++;
382 if (!UTF8_IS_TRAIL(tmp)) return UTF_ILLEGAL;
383 c = (c << 6) | (tmp & 0x3F);
384 /* fall-through */
385 case 2:
386 if (*p == e) return UTF_ILLEGAL;
387 tmp = **p;
388 (*p)++;
389 if (!UTF8_IS_TRAIL(tmp)) return UTF_ILLEGAL;
390 c = (c << 6) | (tmp & 0x3F);
391 /* fall-through */
392 case 1:
393 if (*p == e) return UTF_ILLEGAL;
394 tmp = **p;
395 (*p)++;
396 if (!UTF8_IS_TRAIL(tmp)) return UTF_ILLEGAL;
397 c = (c << 6) | (tmp & 0x3F);
398 }
399
400 /* Check code point validity: no surrogates and valid range */
401 if (!UTF8_IS_VALID(c)) return UTF_ILLEGAL;
402
403 /* make sure it is the most compact representation */
404 if (utf8_width(c) != trail_size + 1) return UTF_ILLEGAL;
405
406 if (html && c < 0xA0) return UTF_ILLEGAL;
407 return c;
408}

418static void
419utf8_encode(unsigned long value, char utf8_seq[4], unsigned *utf8_seqlen)
420{
421 /* struct utf8_seq out={0}; */
422 if (value <= 0x7F) {
423 utf8_seq[0] = value;
424 *utf8_seqlen = 1;
425 }
426 else if (value <= 0x7FF) {
427 utf8_seq[0] = (value >> 6) | 0xC0;
428 utf8_seq[1] = (value & 0x3F) | 0x80;
429 *utf8_seqlen = 2;
430 }
431 else if (value <= 0xFFFF) {
432 utf8_seq[0] = (value >> 12) | 0xE0;
433 utf8_seq[1] = ((value >> 6) & 0x3F) | 0x80;
434 utf8_seq[2] = (value & 0x3F) | 0x80;
435 *utf8_seqlen = 3;
436 }
437 else {
438 utf8_seq[0] = (value >> 18) | 0xF0;
439 utf8_seq[1] = ((value >> 12) & 0x3F) | 0x80;
440 utf8_seq[2] = ((value >> 6) & 0x3F) | 0x80;
441 utf8_seq[3] = (value & 0x3F) | 0x80;
442 *utf8_seqlen = 4;
443 }
444}

446static char *
447utf8_append(unsigned long x, char *d)
448{
449 unsigned utf8_seqlen;
450 char utf8_seq[4];
451 unsigned i;
452
453 utf8_encode(x, utf8_seq, &utf8_seqlen);
454
455 for (i = 0; i < utf8_seqlen; ++i)
456 *d++ = utf8_seq[i];
457 return d;
458}

460size_t
461jsmnf_unescape(char **p_dest, const char src[], size_t size)
462{
463 enum { TESTING = 1, ALLOCATING, UNESCAPING } state = TESTING;
464
465 char *src_start = (char *)src, *src_end = (char *)src + size;
466 char *out_start = NULL, *d = NULL, *s = NULL;
467 unsigned first_surrogate;
468 int second_surrogate_expected;
469 char c;
470
471second_iter:
472 first_surrogate = 0;
473 second_surrogate_expected = 0;
474
475 for (s = src_start; s < src_end;) {
476 c = *s++;
477
478 if (second_surrogate_expected && c != '\\') goto _err;
479 if (0 <= c && c <= 0x1F) goto _err;
480
481 if ('\\' == c) {
482 /* break the while loop */
483 if (TESTING == state) {
484 state = ALLOCATING;
485 break;
486 }
487
488 /* return if src is a well-formed json string */
489 if (s == src_end) goto _err;
490
491 c = *s++;
492
493 if (second_surrogate_expected && c != 'u') goto _err;
494
495 switch (c) {
496 case '"': case '\\': case '/':
497 *d++ = c;
498 break;
499 case 'b': *d++ = '\b'; break;
500 case 'f': *d++ = '\f'; break;
501 case 'n': *d++ = '\n'; break;
502 case 'r': *d++ = '\r'; break;
503 case 't': *d++ = '\t'; break;
504 case 'u': {
505 unsigned x;
506
507 if (!read_4_digits(&s, src_end, &x)) goto _err;
508
509 if (second_surrogate_expected) {
510 if (!UTF16_IS_SECOND_SURROGATE(x)) goto _err;
511
512 d = utf8_append(UTF16_JOIN_SURROGATE(first_surrogate, x),
513 d);
514 second_surrogate_expected = 0;
515 }
516 else if (UTF16_IS_FIRST_SURROGATE(x)) {
517 second_surrogate_expected = 1;
518 first_surrogate = x;
519 }
520 else {
521 d = utf8_append(x, d);
522 }
523 } break;
524 default:
525 goto _err;
526 }
527 }
528 else if (UNESCAPING == state) {
529 *d++ = c;
530 }
531 }
532
533 switch (state) {
534 case UNESCAPING:
535 if (!utf8_validate(out_start, d)) goto _err;
536
537 *p_dest = out_start;
538 return d - out_start;
539 case ALLOCATING:
540 out_start = calloc(1, size);
541 d = out_start;
542 state = UNESCAPING;
543 goto second_iter;
544 case TESTING:
545 *p_dest = calloc(1, size + 1);
546 memcpy(*p_dest, src_start, size);
547 (*p_dest)[size] = '\0';
548 return size;
549 default:
550 break;
551 }
552
553_err:
554 if (UNESCAPING == state) free(out_start);
555 return 0;
556}
